usage: handleregistrartool.sh
 -c,--config-file <arg>   Configuration file. Default is
                          $HOME/.config/handle/handleregistrar.properties
 -h,--help                Print this message
 -q,--query <arg>         iTQL query for getting PIDs from DOMS. Must
                          return a list with PIDs as first element.
 -u,--url-pattern <arg>   URL pattern for what the objects should resolve
                          as. %s is replaced with PID.
